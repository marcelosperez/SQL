-- Problem: Find the names and salary of all employees 
-- who work in the same department as the employee 
-- with the last name 'Grant'
SELECT
  first_name,
  last_name,
  salary
FROM
  hr.employees
WHERE
  department_id IN (
    SELECT
      department_id
    FROM
      hr.employees
    WHERE
      last_name = 'Grant'
  )
AND
  last_name ! = 'Grant';
